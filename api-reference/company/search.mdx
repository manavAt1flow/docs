## Overview

Search for companies using advanced filters including industry, size, location, funding stage, technology, and growth metrics. Perfect for prospecting, market research, competitive analysis, and building targeted company lists.

## Authentication

<ParamField header="X-API-KEY" type="string" required>
  Your Tuesday API key
</ParamField>

## Query Parameters

<ParamField query="industry" type="string">
  Industry or sector filter
  
  **Examples:** `"Technology"`, `"Healthcare"`, `"Finance"`, `"E-commerce"`, `"SaaS"`
</ParamField>

<ParamField query="company_size" type="string">
  Company size range by employee count
  
  **Options:** `"1-10"`, `"11-50"`, `"51-200"`, `"201-500"`, `"501-1000"`, `"1001-5000"`, `"5001+"`
</ParamField>

<ParamField query="location" type="string">
  Geographic location (city, state, country)
  
  **Examples:** `"San Francisco"`, `"California"`, `"United States"`, `"London"`
</ParamField>

<ParamField query="country" type="string">
  Country filter (ISO codes or full names)
  
  **Examples:** `"US"`, `"United States"`, `"UK"`, `"Canada"`, `"Germany"`
</ParamField>

<ParamField query="founded_year_min" type="number">
  Minimum founding year
  
  **Example:** `2015` (companies founded in 2015 or later)
</ParamField>

<ParamField query="founded_year_max" type="number">
  Maximum founding year
  
  **Example:** `2020` (companies founded in 2020 or earlier)
</ParamField>

<ParamField query="funding_stage" type="string">
  Funding stage filter
  
  **Options:** `"Pre-Seed"`, `"Seed"`, `"Series A"`, `"Series B"`, `"Series C"`, `"Series D+"`, `"IPO"`, `"Acquired"`
</ParamField>

<ParamField query="revenue_range" type="string">
  Annual revenue range (USD)
  
  **Options:** `"0-1M"`, `"1M-10M"`, `"10M-50M"`, `"50M-100M"`, `"100M-500M"`, `"500M+"`
</ParamField>

<ParamField query="technology" type="string">
  Technology or tools used by the company
  
  **Examples:** `"Salesforce"`, `"AWS"`, `"React"`, `"HubSpot"`, `"Slack"`
</ParamField>

<ParamField query="keywords" type="string">
  Keywords to search in company descriptions and profiles
  
  **Examples:** `"artificial intelligence"`, `"e-commerce platform"`, `"fintech"`
</ParamField>

<ParamField query="growth_stage" type="string">
  Company growth stage classification
  
  **Options:** `"Startup"`, `"Scale-up"`, `"Growth"`, `"Mature"`, `"Enterprise"`
</ParamField>

<ParamField query="type" type="string">
  Company type filter
  
  **Options:** `"Public"`, `"Private"`, `"Non-profit"`, `"Government"`
</ParamField>

<ParamField query="has_funding" type="boolean">
  Filter for companies that have raised funding
  
  **Options:** `true` | `false`
</ParamField>

<ParamField query="limit" type="number" default="10">
  Maximum number of results to return (1-100)
</ParamField>

<ParamField query="offset" type="number" default="0">
  Number of results to skip for pagination
</ParamField>

<ParamField query="sort_by" type="string" default="relevance">
  Sort order for results
  
  **Options:** `"relevance"`, `"employee_count"`, `"founded_year"`, `"funding_amount"`, `"name"`
</ParamField>

<ParamField query="sort_order" type="string" default="desc">
  Sort direction
  
  **Options:** `"asc"` | `"desc"`
</ParamField>

```python Python
import requests

def search_companies(**filters):
    url = "https://api.tuesday.so/api/v1/company/search"
    headers = {'X-API-KEY': 'your-api-key-here'}
    
    response = requests.get(url, params=filters, headers=headers)
    return response.json()

# Usage Examples
# Find SaaS companies in growth stage
saas_companies = search_companies(
    industry='SaaS',
    growth_stage='Growth',
    company_size='501-1000',
    country='US',
    limit=50
)

# Find well-funded AI startups
ai_startups = search_companies(
    keywords='artificial intelligence machine learning',
    founded_year_min=2018,
    funding_stage='Series B',
    sort_by='funding_amount',
    sort_order='desc'
)

# Find tech companies using specific technology
react_companies = search_companies(
    technology='React',
    industry='Technology',
    company_size='51-200',
    location='California'
)

# Find healthcare companies by revenue
healthcare_companies = search_companies(
    industry='Healthcare',
    revenue_range='10M-50M',
    type='Private',
    sort_by='employee_count'
)

for company in saas_companies['data']:
    print(f"{company['name']} - {company['employee_count']} employees")
    print(f"Founded: {company['founded_year']} | Industry: {company['industry']}")
    if company.get('funding_summary'):
        print(f"Funding: ${company['funding_summary']['total_funding']:,}")
    print("---")
```

function searchCompanies($filters = []) {

## Pagination & Bulk Exports

<CodeGroup>

```javascript Complete Market Export
async function exportMarketSegment(filters, maxResults = 5000) {
  let allCompanies = [];
  let offset = 0;
  const batchSize = 100;
  
  while (allCompanies.length < maxResults) {
    const params = new URLSearchParams({
      ...filters,
      limit: Math.min(batchSize, maxResults - allCompanies.length),
      offset: offset
    });
    
    const response = await fetch(
      `https://api.tuesday.so/api/v1/company/search?${params}`,
      { headers: { 'X-API-KEY': 'your-api-key-here' } }
    );
    
    const data = await response.json();
    
    if (!data.data || data.data.length === 0) break;
    
    allCompanies.push(...data.data);
    
    if (!data.pagination.has_more) break;
    
    offset += batchSize;
    
    // Rate limiting pause
    await new Promise(resolve => setTimeout(resolve, 3000));
  }
  
  return allCompanies.slice(0, maxResults);
}

// Export all SaaS companies in California
const californiaSaas = await exportMarketSegment({
  industry: 'SaaS',
  location: 'California',
  company_size: '51-200,201-500',
  has_funding: true
}, 2000);
```
